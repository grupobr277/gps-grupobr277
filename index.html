// CORREÇÃO 2: RÁDIO PX ROBUSTO let mediaRecorder, chunks = []; const pxBtn = document.getElementById('btn-px'); let gravando = false; async function ligarRadio(e) { e.preventDefault(); e.stopPropagation(); chunks = []; mediaRecorder.ondataavailable = (e) => { if (e.data.size > 0) chunks.push(e.data); }; mediaRecorder.onstop = () => { if (chunks.length === 0) return; const audioBlob = new Blob(chunks, { type: 'audio/webm;codecs=opus' }); const reader = new FileReader(); reader.onloadend = () => { // Envia apenas se tiver conteúdo if (reader.result && reader.result.length > 1000) { db.ref("radio_v25").set({ audio: reader.result, u: user, t: Date.now() }); } // Libera recursos stream.getTracks().forEach(track => track.stop()); }; reader.readAsDataURL(audioBlob); chunks = []; gravando = false; pxBtn.innerHTML = '🎤'; }; mediaRecorder.start(); } catch(err) { console.log("Erro microfone:", err); alert("Permita o acesso ao microfone para usar o rádio!"); pxBtn.classList.remove('ativo'); pxBtn.innerHTML = '🎤'; gravando = false; } } function desligarRadio(e) { e.preventDefault(); e.stopPropagation(); if (mediaRecorder && mediaRecorder.state === "recording") { mediaRecorder.stop(); } else { pxBtn.classList.remove('ativo'); pxBtn.innerHTML = '🎤'; gravando = false; } } // CORREÇÃO 3: EVENTOS MAIS CONFIÁVEIS pxBtn.addEventListener('touchstart', ligarRadio, { passive: false }); pxBtn.addEventListener('touchend', desligarRadio, { passive: false }); pxBtn.addEventListener('touchcancel', desligarRadio, { passive: false }); pxBtn.addEventListener('mousedown', ligarRadio); pxBtn.addEventListener('mouseup', desligarRadio); pxBtn.addEventListener('mouseleave', desligarRadio); // CORREÇÃO 4: RECEPÇÃO DE ÁUDIO MAIS ESTÁVEL db.ref("radio_v25").on("value", snap => { const d = snap.val(); if (d && d.u !== user && !isMuted && (Date.now() - d.t < 5000)) { try { const audio = new Audio(d.audio); audio.volume = 0.8; audio.play().catch(e => console.log("Erro ao reproduzir:", e)); } catch(e) { console.log("Áudio inválido"); } } }); // CORREÇÃO 5: BOTÃO DE RECENTRALIZAR MAIS INTUITIVO function recentralizar() { follow = true; if (rotaControle) map.removeControl(rotaControle); if (meuLatLng) { rotaControle = L.Routing.control({ waypoints: [meuLatLng, dest], lineOptions: { styles: [{ color: '#007aff', weight: 7, opacity: 0.8 }] }, createMarker: () => null, show: false, addWaypoints: false }).addTo(map); } else { alert("Aguardando localização..."); } }).addTo(map); } }, 2000);
